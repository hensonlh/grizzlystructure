qmatrixNoKiti<-read.csv('QmatrixNoKiti.csv') structnokitimatrix<-as.matrix(dist(qmatrixNoKiti, method = "euclidean"))langfactor<-read.csv("LangFactorNoKitiJustFactor.csv")langfactorunlist<-unlist(langfactor)langfactordf<-as.data.frame(langfactorunlist)class(langfactordf)class(langfactorunlist)langanosimtest<-anosim(structnokitimatrix, langfactorunlist, permutations = 999, strata = NULL)summary(langanosimtest)##############################################################################################MANOVAqvalues<-read.csv("Qvalues.csv",
